void initializePWMTimer() {
  cli();
  TIMER2_PWM_A_PIN_MODE_OUTPUT(); // Set the A PWM pin of TIMER2 to output
  ASSR &= ~(_BV(EXCLK) | _BV(AS2));
  TCCR2A = (1 << COM2A0) | (1 << WGM21) | (1 << WGM20);
  TCCR2B = (1 << WGM22) | (1 << CS20);
  OCR2A = 0;
  sei();
}

byte readCameraRegister(byte registerId) {
  Wire.beginTransmission(CAMERA_ADDRESS);
  Wire.write(registerId);
  Wire.endTransmission();
  Wire.requestFrom(CAMERA_ADDRESS, 1);
  while (Wire.available() <= 0);
  byte registerValue = Wire.read();
  delay(1);
  return registerValue;
}

void writeCameraRegister(byte registerId, byte registerValue) {
  Wire.beginTransmission(CAMERA_ADDRESS);
  Wire.write(registerId);
  Wire.write(registerValue);
  Wire.endTransmission();
  delay(1);
}

void captureFrame(unsigned int frameWidth, unsigned int frameHeight) {
  Serial.print("*RDY*"); // send to the frame capture software a "start of frame" message for beginning capturing

  delay(1000);

  cli(); // disable all interrupts during frame capture (because it needs to be as fast as possible)
  while (PIN3_DIGITAL_READ() == LOW); // wait until VS/VSYNC pin is high
  while (PIN3_DIGITAL_READ() == HIGH); // wait until VS/VSYNC pin is low
  unsigned int tempWidth = 0;
  while (frameHeight--) {
    tempWidth = frameWidth;
    while (tempWidth--) {
      while (PIN2_DIGITAL_READ() == LOW); // wait until PCLK pin is high
      while (PIN2_DIGITAL_READ() == HIGH); // wait until PCLK pin is low
      byte byteToWrite = 0b00000000;
      byteToWrite |= ((PIN7_DIGITAL_READ() == HIGH) << 7);
      byteToWrite |= ((PIN6_DIGITAL_READ() == HIGH) << 6);
      byteToWrite |= ((PIN5_DIGITAL_READ() == HIGH) << 5);
      byteToWrite |= ((PIN4_DIGITAL_READ() == HIGH) << 4);
      byteToWrite |= ((PINA3_DIGITAL_READ() == HIGH) << 3);
      byteToWrite |= ((PINA2_DIGITAL_READ() == HIGH) << 2);
      byteToWrite |= ((PINA1_DIGITAL_READ() == HIGH) << 1);
      byteToWrite |= ((PINA0_DIGITAL_READ() == HIGH));
      UDR0 = byteToWrite; // send data via serial connection with UART register (we need to use the serial register directly for fast transfer)
      while (PIN2_DIGITAL_READ() == LOW); // wait until PCLK pin is high
      while (PIN2_DIGITAL_READ() == HIGH); // wait until PCLK pin is low
      // ignore each second byte (for a grayscale image we only need each first byte, which represents luminescence)
    }
  }
  sei(); // enable all interrupts
  
  delay(1000);
}


void setup() {
  initializePWMTimer();
  Wire.begin();
  Serial.begin(1000000); // the frame capture software communicates with the Arduino at a baud rate of 1MHz
}

void loop() {
  captureFrame(320, 240); // capture a frame at QVGA resolution (320 x 240)
}

30August2019 - Bitluni ULP Sound Driver
=======================================
1. ULPSoundMono.ino : 

    '0x1D4C0121' hex value is '0b11101010011000000000100100001' in binary.
    
    This value is used to create a custom REG_WR [TRM: page612] ULP Co-processor instruction to 
    populate the opcode table invented by Bitluni in the C-macros ulp_insn_t program named 'mono' 
    in this arduino file.

2. '0b11101010011000000000100100001' binary can be broken down into 5 sections:

    1) first 10bits (counting from LSB) - addr of the register to write data to
                                        - '0100100001' binary value is a 32bit word addr_ulp, 
                                          referring to 'RTCIO_PAD_DAC1_REG' register as seen from ULP.
                                          To convert this to the address as seen from main CPU, use this formula:
                                          
                                          addr_ulp = (addr_dport - DR_REG_RTCCNTL_BASE)/4    [TRM: page612]
                                          
                                          addr_dport is the address as seen from main CPU that we're looking for.
                                          so, 
                                          
                                          addr_dport = addr_ulp * 4 + DR_REG_RTCCNTL_BASE = 0x3FF48484
                                          
                                          [DR_REG_RTCCNTL_BASE(0x3ff48000) define macro in 'esp-idf/components/soc/esp32/include/soc/soc.h']
                                          
                                        - '0x3FF48484' hex value refers to 'RTCIO_PAD_DAC1_REG' register. [TRM: page78]
                                          
                                          The lower & upper 5 bits in point 3) & 4) below respectively refers to 
                                          the 'RTCIO_PAD_PDAC1_DAC' bit in 'RTCIO_PAD_DAC1_REG' register.
                                          'RTCIO_PAD_PDAC1_DAC' bit is the PAD DAC1 output value.
                                          
                                          ['RTC_IO_PAD_DAC1_REG' define macro is defined as '(DR_REG_RTCIO_BASE + 0x84)'
                                          in 'esp-idf/components/soc/esp32/include/soc/rtc_io_reg.h']
        
    2) next 8bits                       - data value to be written to the addr in 1)
    3) next 5bits                       - lower bit of the register pointed by addr in 1)
    4) next 5bits                       - upper bit of the register pointed by addr in 1)
    6) last 1bit                        - opcode referring to REG_WR instruction in ULP Co-processor.
                                          [OPCODE_WR_REG define macro in 'esp-idf/components/ulp/include/esp32/ulp.h']

3.  So, we're using ULP Co-processor to output the analog voltage output directly to dacn_out.
    DAC data in dacn_out can be connected to I2S_DATA_OUT directly(for direct memory access) to be
    output by I2S via DMA.  [TRM: page585]
    
    As a result, we can use ULP Co-processor to output audio data directly to dacn_out which is
    connected to I2S which then output the audio data via DMA to external speaker, all without 
    direct main CPU intervention, thus saving on CPU cycles for other cpu intensive work.  [TRM: page586]
    
    All main CPU has to do is to time to time copy the audio data to the data section in RTC_SLOW_MEM.
    Each sample of audio data correspond to a certain level of DAC output voltage amplitude(1-256 in Bitluni's example)
    which is imbued into the 256 custom REG_WR instructions in the opcode table invented by Bitluni in the mono ulp_insn_t program.
    So, every sample will pick its corresponding DAV output voltage amplitude and the custom REG_WR instruction will 
    write it directly to the dacn_out register that connects directly to I2S_DATA_OUT.
